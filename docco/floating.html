      if (x === Infinity) {
        return 1;
      } else if (x === -Infinity) {
        return -1;
      } else {
        return (Math.exp(x) - Math.exp(-x)) / (Math.exp(x) + Math.exp(-x));
      }
    },

    cosh: Math.cosh || function (x) {
      return (Math.exp(x) + Math.exp(-x)) / 2;
    },

        return x;
      } else {
        return Math.log(x + Math.sqrt(x * x + 1));
      }
    },

    atanh: Math.atanh || function (x) {
      return Math.log((1 + x) / (1 - x)) / 2;
    },

      return Math.log(x + Math.sqrt(x * x - 1));
    }

  };
